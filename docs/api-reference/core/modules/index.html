Function

createFunction

Description

AI가 호출할 수 있는 함수를 생성합니다. 함수 이름, 설명, 매개변수 스키마, 실행 로직을 정의할 수 있습니다.

Example

import { z } from 'zod';
import { createFunction } from '@robota/core';

const getWeather = createFunction({
  name: 'getWeather',
  description: '특정 위치의 날씨 정보를 조회합니다.',
  parameters: z.object({
    location: z.string().describe('날씨를 조회할 위치 (도시명)'),
    unit: z.enum(['celsius', 'fahrenheit']).optional().describe('온도 단위')
  }),
  execute: async (params) => {
    // 날씨 API 호출 로직
    return { temperature: 25, condition: '맑음' };
  }
});
